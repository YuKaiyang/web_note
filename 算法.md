1. Smallest Common Multiple (计算任意两正整数之间所有整数的最小公倍数)
```
function smallestCommons(arr) {
  function org(arr) {
    let arr2 = arr.concat()
    arr2.sort((a, b) => {
      return a > b
    })
    const min = arr2[0]
    const max = arr2[1]
    arr2 = []

    for (let i = min; i <= max; i++) {
      arr2.push(i)
    }
    return arr2
  }

  const arr2 = org(arr)
  const len = arr2.length
  let res = arr2[len - 1]

  arr2.forEach(v => {
    let i = 1
    while (res * i % v !== 0) {
      i++
    }
    res *= i
  })

  return res
}
```
